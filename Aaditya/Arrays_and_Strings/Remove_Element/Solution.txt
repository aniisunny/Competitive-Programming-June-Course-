class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int valCount =0;
        for(int i =0;i<nums.size();i++){
            if(nums[i]==val){
                valCount++;
            }
            else{
            nums[i-valCount]= nums[i];
        }
        }
        
        return nums.size()- valCount;
    }
};
